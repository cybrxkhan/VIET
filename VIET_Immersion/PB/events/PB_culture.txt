# Province culture changes
province_event = {
	id = 55000
	desc = "EVTDESC55000"
	picture = GFX_evt_stone_church
	
	trigger = {
		owner = {
			NOT = { culture = ROOT }
			religion = ROOT
		}
		any_neighbor_province = {
			has_owner = yes
			owner = { culture = PREV }
			ROOT = {
				owner = { culture = PREVPREV }
			}
		}
		NOT = { # No flipping back to Norse culture before 1150
			AND = {
				NOT = {	has_global_flag = norse_culture_flipping }
				owner = { culture = norse }
				ROOT = {
					OR = {
						culture = swedish
						culture = norwegian
						culture = danish
						culture = norman
						culture = icelandic
						culture = norsegael
						culture = norsebriton
					}
				}
			}
		}
		#These setups use melting pot events instead
		NOT = {
			AND = {
				culture_group = celtic
				owner = { culture_group = north_germanic }
			}
		}
		NOT = {
			AND = {
				OR = {
					culture = italian
					culture = greek
					culture = saxon
					culture = siculo_arabic
				}
				owner = { culture = norman }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 3000

		modifier = {
			factor = 0.75
			port = yes
		}
		modifier = {
			factor = 0.4
			NOT = {
				any_neighbor_province = { culture = ROOT }
			}
		}
		modifier = {
			factor = 1.5
			owner = { NOT = { stewardship = 2 }}
		}
		modifier = {
			factor = 1.5
			owner = { NOT = { stewardship = 3 }}
		}
		modifier = {
			factor = 1.5
			owner = { NOT = { stewardship = 4 }}
		}
		modifier = {
			factor = 1.5
			owner = { NOT = { stewardship = 5 }}
		}
		modifier = {
			factor = 0.84
			owner = { stewardship = 10 }
		}
		modifier = {
			factor = 0.84
			owner = { stewardship = 11 }
		}
		modifier = {
			factor = 0.84
			owner = { stewardship = 12 }
		}
		modifier = {
			factor = 0.84
			owner = { stewardship = 13 }
		}
		modifier = {
			factor = 0.4
			owner = { culture = mongol }
			NOT = { year = 1285 }
		}
		modifier = {
			factor = 0.75
			owner = {
				OR = {
					primary_title = {
						conquest_culture = PREV
					}
					ROOT = {
						county = {
							conquest_culture = PREVPREV
						}
					}
					ROOT = {
						duchy = {
							holder_scope = {
								character = PREVPREVPREV
							}
							conquest_culture = PREVPREV
						}
					}
					ROOT = {
						kingdom = {
							holder_scope = {
								character = PREVPREVPREV
							}
							conquest_culture = PREVPREV
						}
					}
					ROOT = {
						empire = {
							holder_scope = {
								character = PREVPREVPREV
							}
							conquest_culture = PREVPREV
						}
					}
				}
			}
		}
		#Turkish culture didn't spread to coasts much

			factor = 3
			port = yes
			owner = {
				culture = turkish

		}

			factor = 0.3

				culture = turkish

			factor = 0.66
