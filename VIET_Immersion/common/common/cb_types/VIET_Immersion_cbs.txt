turkish_conquest = {
	name = CB_NAME_TURKISH_CONQUEST
	war_name = WAR_NAME_TURKISH_CONQUEST
	sprite = 8
	truce_days = 3650
	hostile_against_others = yes
	is_permanent = yes
	is_holy_war = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = DUKE # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	can_use_gui = {
		ROOT = {
			piety = 100
			OR = {
				trait = ambitious
				trait = diligent
				trait = wroth
				trait = envious
				trait = brave
				trait = proud
				trait = zealous
				AND = {
					NOT = { trait = content }
					NOT = { trait = slothful }
					NOT = { trait = imbecile }
					NOT = { trait = craven }
				}
			}
			OR = {
				NOT = { has_character_modifier = holy_war_cooldown }
				has_global_flag = cb_cooldowns_off
			}			
		}
	}
	
	can_use = {
		ROOT = {
			NOT = { has_global_flag = VIET_turkish_features_off }
			religion_group = muslim
			OR = {
				culture = turkish
				has_landed_title = k_rum
				has_landed_title = e_turkish
			}
		}
		
		FROM = { 
			NOT = { religion = ROOT }
			independent = yes
			OR = {
				has_landed_title = e_byzantium
				AND = {
					culture = greek
					primary_title = { de_jure_liege_or_above = e_byzantium }
				}
				AND = {
					OR = {
						religion = orthodox
						is_heresy_of = orthodox
					}
					ROOT = { has_landed_title = e_turkish }
				}
			}
		}
		
		ROOT = {
			NOT = { is_liege_or_above = FROM }
			NOT = { distance = { who = FROM distance = 150 } }
		}
	}
	
	can_use_title = {
		# Can't attack Constantinople unless Byzantine Empire has less than 30 holdings
		OR = {
			NOT = { title = d_thrace }
			e_byzantium = {
				holder_scope = {
					NOT = { realm_size = 30 }
				}
			}
		}
	}
	
	on_add = {
		ROOT = { piety = -100 }
		ROOT = {
			add_character_modifier = {
				name = holy_war_cooldown
				duration = 1825
			}
		}
		ROOT = {
			hidden_tooltip = {
				if = {
					limit = {
						NOT = { has_character_flag = turkish_got_free_troops_already }
					}
					character_event = { id = ottoman.3 }
				}
			}
		}
	}

	on_success = {
		ROOT = {
			prestige = 200
			if = {
				limit = {
					has_character_flag = turkish_got_free_troops_already
				}
				clr_character_flag = turkish_got_free_troops_already
			}
		}
		
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 200
			
			if = {
				limit = {
					religion_group = muslim
					FROM = { NOT = { religion_group = muslim } }
				}
				participation_scaled_decadence = -20
			}
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 200 
				if = {
					limit = {
						religion_group = muslim
						FROM = { NOT = { religion_group = muslim } }
					}
					participation_scaled_decadence = -20
				}
			}
		}
		FROM = {
			prestige = -200
		}
	}
	
	on_success_title = {
		custom_tooltip = {
			text = turkish_conquest_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
						is_crusade = yes # Even if the title holder is not participating in the war, gain holdings occupied by all Crusade participants
					}
				}
			}
		}
	}

	on_fail = {
		FROM = {
			prestige = 100
		}
#		FROM = {
#			if = {
#				limit = {
#					has_dlc = "Legacy of Rome"
#					OR = { 
#						has_landed_title = e_byzantium
#						has_landed_title = e_roman_empire
#					}
#					religion_group = christian
#				}
#				hidden_tooltip = { character_event = { id = LoR.30 days = 12 } }
#			}
#		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		ROOT = {
			prestige = -200
			if = {
				limit = {
					has_character_flag = turkish_got_free_troops_already
				}
				clr_character_flag = turkish_got_free_troops_already
			}
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			decadence = 10
		}
		FROM = {
			prestige = 200
		}
#		FROM = {
#			if = {
#				limit = { 
#					OR = { 
#						has_landed_title = e_byzantium
#						has_landed_title = e_roman_empire
#					}
#					religion_group = christian
#				}
#				hidden_tooltip = { character_event = { id = LoR.30 days = 12 } }
#			}
#		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 200
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
		ROOT = {
			if = {
				limit = {
					has_character_flag = turkish_got_free_troops_already
				}
				clr_character_flag = turkish_got_free_troops_already
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = 100
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 150
	}
	
	ai_will_do = { 
		factor = 1
		modifier = {
			factor = 2
			ROOT = {
				OR = {
					trait = ambitious
					trait = zealous
					trait = diligent
					trait = greedy
				}
			}
		}
		modifier = {
			factor = 2.5
			NOT = { year = 1150 }
			ROOT = {
				OR = { 
					has_landed_title = k_rum
					liege = { has_landed_title = k_rum }
					top_liege = { has_landed_title = k_rum }
				}
			}
		}
		modifier = {
			factor = 2
			year = 1150
			ROOT = {
				OR = { 
					has_landed_title = k_rum
					liege = { has_landed_title = k_rum }
					top_liege = { has_landed_title = k_rum }
				}
			}
		}
		modifier = {
			factor = 5
			ROOT = {
				OR = { 
					has_landed_title = e_turkish
					liege = { has_landed_title = e_turkish }
					top_liege = { has_landed_title = e_turkish }
				}
			}			
		}
	}
}

hypocrite_county_conquest = { 
	name = CB_NAME_HYPOCRITE_CONQUEST
	war_name = WAR_NAME_HYPOCRITE_CONQUEST
	sprite = 16
	truce_days = 1825
	hostile_against_others = yes
	is_permanent = yes
	check_all_titles = yes # if permanent, setting this to true will check against all of someones titles, including vassal held titles
	press_claim = yes
	can_ask_to_join_war = no
	
	can_use_gui = {
		ROOT = {
			OR = {
				trait = ambitious
				trait = diligent
				trait = wroth
				trait = brave
				trait = zealous
			}
			OR = {
				AND = {
					trait = ambitious
					OR = {
						trait = diligent
						trait = wroth
						trait = brave
						trait = zealous					
					}
				}
				AND = {
					trait = zealous
					OR = {
						trait = diligent
						trait = wroth
						trait = brave
						trait = ambitious					
					}
				}
				AND = {
					trait = wroth
					OR = {
						trait = diligent
						trait = zealous
						trait = brave
						trait = ambitious					
					}
				}
				AND = {
					trait = brave
					OR = {
						trait = diligent
						trait = zealous
						trait = wroth
						trait = ambitious					
					}
				}
				AND = {
					trait = zealous
					OR = {
						trait = diligent
						trait = brave
						trait = wroth
						trait = ambitious					
					}
				}
				AND = {
					NOT = { trait = slothful }
					NOT = { trait = imbecile }
					NOT = { trait = cynical }
					NOT = { trait = content }
				}
			}
			prestige = 50
			OR = {
				NOT = { has_character_modifier = holy_war_cooldown }
				has_global_flag = cb_cooldowns_off
			}
		}
		FROM = {
			AND = {
				NOT = { trait = charitable }
				NOT = { trait = kind }
				NOT = { trait = temperate }
				NOT = { trait = zealous }
				NOT = {
					AND = {
						tier = count
						piety = 400
					}
					AND = {
						tier = duke
						piety = 800
					}
					AND = {
						tier = king
						piety = 1600
					}
					AND = {
						tier = emperor
						piety = 3200
					}
				}
					OR = {
						trait = gluttonous
						trait = proud
						trait = wroth
						trait = lustful
						trait = slothful
						trait = greedy
						trait = arbitrary
						trait = envious
						trait = cruel
					}
			}
		}
	}
	
	on_add = {
		ROOT = { 
			prestige = -50
			add_character_modifier = {
				name = holy_war_cooldown
				duration = 1825
			}			
		}
	}
	
	can_use = {
		ROOT = {
			NOT = { is_liege_or_above = FROM }
			religion = dulcinian
		}
		FROM = {
			religion_group = christian
		}
	}

	can_use_title = {
		tier = count
		OR = {
			FROM = {
				has_landed_title = PREV
			}
			holder_scope = {
				is_liege_or_above = FROM
				NOT = { same_realm = ROOT }
			}
		}
		
		ROOT = {
			religion_group = christian
		}

		NOT = {
			OR = {
				ROOT = {
					has_landed_title = PREV
				}
				holder_scope = {
					is_liege_or_above = ROOT
				}
			}
		}
		
		location = {
			any_neighbor_province = {
				owner = {
					OR = {
						character = ROOT
						is_liege_or_above = ROOT
					}
				}
			}
		}
	}

	is_valid_title = {
		OR = {
			FROM = {
				has_landed_title = PREV
			}
			holder_scope = {
				is_liege_or_above = FROM
			}
		}
	}

	on_success_title = {
		if = {
			limit = {
				holder_scope = {
					tier = count
					NOT = { num_of_count_titles = 2 }
					lower_tier_than = ROOT
				}
			}

			holder_scope = {
				set_defacto_liege = ROOT
			}
		}

		if = {
			limit = {
				holder_scope = {
					OR = {
						higher_tier_than = count
						num_of_count_titles = 2
						NOT = { lower_tier_than = ROOT }
					}
				}
			}

			usurp_title_plus_barony_if_unlanded = ROOT
			any_de_jure_vassal_title = { # take all baronies under the one we're fighting for
				limit = {
					has_holder = yes
					NOT = {
						de_facto_liege = PREV
					}
					holder_scope = {
						OR = {
							character = FROM
							is_liege_or_above = FROM
						}
					}
				}

				usurp_title_plus_barony_if_unlanded = ROOT
			}
		}
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_piety = 100
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { participation_scaled_piety = 100 }
		}
	}

	on_fail_title = {
		ROOT = {
			prestige = -100
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 50
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 50 }
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	# ai importance placed on this CB: scope is the targeted title, ROOT is the attacking character, FROM is the defending character
	ai_will_do = { 
		factor = 1
		modifier = {
			factor = 0.75 #slow down
		}
		modifier = {
			factor = 0.05
			AND = {
				ROOT = { culture_group = east_african }
				FROM = { culture_group = east_african }
			}
		}
	}
}

khitan_invasion = {
	name = CB_NAME_INVASION
	war_name = WAR_NAME_INVASION
	sprite = 8
	truce_days = 365
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	can_use = {
		ROOT = {
			has_landed_title = e_khitan
			culture = khitan
			NOT = { year = 1170 }
			NOT = { realm_size = 80 }
		}
	}
	
	can_use_title = {
		ROOT = {
			NOT = { is_liege_or_above = FROM }
		}
	}
	
	is_valid_title = {
		FROM = {
			any_realm_title = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	on_success = {
		ROOT = {
			prestige = 200
		}
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 200
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
		FROM = {
			prestige = -200
		}
		hidden_tooltip = {
			ROOT = {
				occupy_minors_of_occupied_settlements = FROM
				gain_all_occupied_titles = FROM
			}
		}
	}
	
	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				if = {
					limit = {
						has_holder = yes
						ROOT = { tier = emperor }
						NOT = { FROM = { tier = emperor } }
						NOT = { FROM = { tier = king } }
					}
					holder_scope = {
						set_defacto_liege = ROOT
					}
				}
				any_de_jure_vassal_title = {
					limit = {
						holder_scope = {
							independent = no
							liege = {
								character = PREV
							}
							any_war = {
								defender = {
									character = FROM
								}
							}
						}
					}
					holder_scope = {
						set_defacto_liege = ROOT
					}
				}
			}
		}
	}

	on_fail = {
		FROM = {
			prestige = 100
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		ROOT = {
			prestige = -200
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		FROM = {
			prestige = 200
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 200
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
	}

	attacker_ai_victory_worth = {
		factor = 200
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	# ai importance placed on this CB: scope is the targeted title, ROOT is the attacking character, FROM is the defending character
	ai_will_do = { 
		factor = 1
		modifier = { #Nudge them in the right direction
			factor = 10
			AND = {
				ROOT = {
					has_landed_title = e_khitan
				}
				OR = {
					FROM = {
						any_realm_title = {
							de_jure_liege_or_above = e_persia
						}
					}
					FROM = {
						any_realm_title = {
							de_jure_liege_or_above = k_turkestan
						}
					}
				}
			}		
		}
	}
}
