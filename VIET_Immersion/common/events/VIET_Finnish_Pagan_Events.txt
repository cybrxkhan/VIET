					50 = { }

						character_event = { id = finnish.25 days = 2 random = 3 }
						set_character_flag = bear_hunt_girl_event_happened
					}
				}
			}
		}
		if = { #Be a douchebag to her
			limit = {
				has_character_flag = bear_hunt_girl_be_a_douche
				OR = {
					trait = cruel

					trait = arbitrary
					trait = greedy
				}
