# Israel conquered by non-Jews
narrative_event = {
	id = VIETjewish.1
	title = VIETjewish.1.name
	desc = VIETjewish.1.desc
	picture = GFX_evt_holy_jerusalem
	
	only_rulers = yes
	
	trigger = {
		has_landed_title = k_israel
		NOT = { religion_group = jewish_group }
		has_global_flag = jewish_homeland_restored
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = VIETjewish.1.a
		if = {
			limit = {
				NOT = { has_landed_title = k_jerusalem }
			}
			k_jerusalem = {
				grant_title = ROOT
			}
		}
		k_israel = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_jerusalem
			}
		}
		k_jerusalem = {
			de_jure_liege = e_arabia
		}
		k_israel = {
			destroy_landed_title = THIS
		}
	}
}
