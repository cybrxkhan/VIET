#MIZRAHI
# Jewish Diplomat Appears at Court (Mizrahi)
character_event = {
	id = mizrahi.0
	desc = EVTDESC_SoA_200
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	prisoner = no
	only_playable = yes
	capable_only = yes
	
	trigger = {
		has_dlc = "Sons of Abraham"
		independent = yes
		NOT = { religion_group = pagan_group }
		capital_scope = {
			OR = {
				empire = { title = e_arabia }
				empire = { title = e_persia }
				kingdom = { title = k_africa }
				kingdom = { title = k_numidia }
				kingdom = { title = k_egypt }
			}
		}		
		NOT = {
			has_character_modifier = expelled_jewish
			trait = incapable
		}
	}
	
	mean_time_to_happen = {
		months = 1000
	}
	
	option = {
		name = EVTOPTA_SoA_200
		create_character = {
			random_traits = yes
			culture = mizrahi
			dynasty = culture
			religion = jewish
			female = no
			trait = charismatic_negotiator
			attributes = {
				diplomacy = 10
			}
			flag = ai_flag_refuse_conversion
		}
	}
	option = {
		name = "reject.VIET.courtiers"
		ai_chance = {
			factor = 1
			modifier = {
				factor = 3
				trait = zealous
				NOT = { religion_group = jewish_group }
			}
			modifier = {
				factor = 2
				any_courtier = {
					religion_group = jewish_group
				}
			}
		}
	}
}
