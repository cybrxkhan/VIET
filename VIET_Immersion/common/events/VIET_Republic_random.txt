# Family member marries childhood friend (AI)
character_event = {
	id = VIETrepublic.0
	hide_window = yes
	
	ai = yes
	capable_only = yes
	prisoner = no
	min_age = 18
	only_men = yes
	
	trigger = {
		always = no #not finished yet
		is_married = no
		is_betrothed = no
		is_primary_heir = no
		is_pretender = no
		is_theocracy = no
		is_priest = no
		liege = {
			NOT = { any_child = { character = ROOT } }
			is_patrician = yes
			dynasty = ROOT
		}
		NOT = { age = 30 }
	}
	
	mean_time_to_happen = {
		months = 300
	}
	
	immediate = {
		if = {
			limit = {
				age = 18
				NOT = { age = 22 }
			}
			liege = {
				character_event = { id = VIETrepublic.1 }
			}
		}
		if = {
			limit = {
				age = 22
				NOT = { age = 26 }
			}
			liege = {
				character_event = { id = VIETrepublic.2 }
			}
		}
		if = {
			limit = {
				age = 26
				NOT = { age = 30 }
			}
			liege = {
				character_event = { id = VIETrepublic.3 }
			}
		}
	}
	
	option = {
		name = OK
	}
}
