# Sect event chain start - never had a sect before
character_event = {
	id = buddhist.2
	desc = buddhist.2.desc	
	picture = GFX_evt_buddhist_pagoda
	
	only_rulers = yes
	
	trigger = {
		religion = buddhist
		NOT = { has_character_flag = buddhist_sect_already_chose_once }
		NOT = { has_character_flag = choosing_buddhist_sect }
		primary_title = { higher_tier_than = BARON }
		OR = {
			trait = mahayana_buddhist
			trait = vajrayana_buddhist
			trait = theravada_buddhist
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_character_flag = choosing_buddhist_sect
		set_character_flag = buddhist_sect_already_chose_once
		hidden_tooltip = {
			add_character_modifier = {
				name = recently_chose_buddhist_sect
				duration = 1095
				hidden = yes
			}
		}
	}
	
	option = { #I have decided
		name = buddhist.2.a
		character_event = { id = buddhist.10 tooltip = "buddhist.10.tooltip" }
	}	
	option = { #want to learn more first
		name = buddhist.2.b
		trigger = {
			ai = no
		}
		set_character_flag = learn_more_buddhist_sects
		if = {
			limit = {
				trait = mahayana_buddhist
			}
			character_event = { id = buddhist.5 tooltip = "buddhist.5.tooltip" }
		}
		if = {
			limit = {
				trait = vajrayana_buddhist
			}
			character_event = { id = buddhist.21 tooltip = "buddhist.21.tooltip" }
		}
		if = {
			limit = {
				trait = theravada_buddhist
			}
			character_event = { id = buddhist.26 tooltip = "buddhist.26.tooltip" }
		}			
	}
}
