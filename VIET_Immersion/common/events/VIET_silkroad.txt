# Become SarTA PAAAWWW
character_event = {
	id = silkroad.0
	desc = silkroad.0.desc

	hide_window = yes

	min_age = 16
	capable_only = yes
	prisoner = no
	only_men = yes

	trigger = {
		culture = sogdian
		NOT = { has_global_flag = silk_road_level_0 }
		NOT = { trait = sartapaw }
		OR = {
			is_ruler = no
			tier = baron
			tier = count
			AND = {
				tier = duke
				independent = no
			}
		}
		AND = {
			stewardship = 10
			diplomacy = 10
		}
		OR = {
			stewardship = 13
			diplomacy = 13
		}
	}
	
	mean_time_to_happen = {
		months = 840
		modifier = {
			factor = 0.95
			stewardship = 16
		}
		modifier = {
			factor = 0.95
			diplomacy = 16
		}
		modifier = {
			factor = 0.95
			trait = greedy
		}
		modifier = {
			factor = 1.1
			trait = charitable
		}
		modifier = {
			factor = 0.95
			trait = genius
		}
		modifier = {
			factor = 1.1
			trait = imbecile
		}
		modifier = {
			factor = 0.95
			trait = quick
		}
		modifier = {
			factor = 1.1
			trait = slow
		}
		modifier = {
			factor = 0.95
			trait = diligent
		}
		modifier = {
			factor = 1.1
			trait = slothful
		}
		modifier = {
			factor = 0.95
			trait = ambitious
		}
		modifier = {
			factor = 1.1
			trait = content
		}
		modifier = {
			factor = 0.95
			trait = just
		}
		modifier = {
			factor = 1.1
			trait = arbitrary
		}
	}
	
	immediate = {
		if = {
			limit = {
				ai = yes
			}
			add_trait = sartapaw
		}
		if = {
			limit = {
				ai = no
				NOT = { has_global_flag = silk_road_level_3 }
			}
			character_event = { id = silkroad.1 }
		}
		if = {
			limit = {
				ai = no
				has_global_flag = silk_road_level_3
			}
			character_event = { id = silkroad.2 }
		}
	}
}

		give_nickname = nick_the_liberator
		if = {
			limit = {
				NOT = { has_landed_title = k_sogdiana }
			}
			k_sogdiana = {
				grant_title = ROOT
			}
		}
		d_samarkand = {
			de_jure_liege = k_sogdiana
		}
		d_khiva = {
			de_jure_liege = k_sogdiana
		}

			de_jure_liege = k_sogdiana
		}

		}
