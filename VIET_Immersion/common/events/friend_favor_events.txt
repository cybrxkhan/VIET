# liege responds to request for help with getting married
character_event = {
	id = friendfavors.1
	desc = friendfavors.1.desc
	picture = GFX_evt_marriage
	
	is_triggered_only = yes
	
	option = {
		name = "friendfavors.1.a"  # I have a courtier 
		trigger = {
			FROM = {
				is_married = no
			}
			any_courtier = {
				OR = {
					is_female = no
					NOT = { age = 45 }
				}
				NOT = { is_close_relative = ROOT }
				can_marry = FROM
			}
		}
		ai_chance = { factor = 50 }
		random_courtier = {
			limit = {
				OR = {
					is_female = no
					NOT = { age = 45 }
				}
				NOT = { is_close_relative = ROOT }
				can_marry = FROM
			}
			add_spouse = FROM
		}
		FROM = {
			opinion = {
				modifier = opinion_helpful_friend
				who = ROOT
				years = 10
			}
		}
	}
	
	option = {
		name = "friendfavors.1.b"  # I'll find one
		trigger = {
			NOT = { intrigue = 10 }
			FROM = { is_female = no }
		}
		ai_chance = { factor = 25 }
		scaled_wealth = -0.1
		create_character = {
			female = yes
			random_traits = yes
			dynasty = random
		}
		new_character = {
			add_spouse = FROM
		}
		FROM = {
			opinion = {
				modifier = opinion_helpful_friend
				who = ROOT
				years = 10
			}
		}
	}
	
	option = {
		name = "friendfavors.1.c"  # I'll find an attractive one
		trigger = {
			intrigue = 10
			FROM = { is_female = no }
		}
		ai_chance = { factor = 50 }
		scaled_wealth = -0.075
		create_character = {
			female = yes
			random_traits = yes
			dynasty = random
		}
		new_character = {
			add_trait = fair
			add_spouse = FROM
		}
		FROM = {
			opinion = {
				modifier = opinion_helpful_friend
				who = ROOT
				years = 10
			}
		}
	}
	
	option = {
		name = "friendfavors.1.d"  # Sorry, can't help
		ai_chance = { factor = 0 }
		FROM = {
			opinion = {
				modifier = opinion_unhelpful_friend
				who = ROOT
				years = 15
			}
		}
	}
}

# liege responds to request to stop inter-realm war
character_event = {
	id = friendfavors.14
	desc = friendfavors.14.desc
	picture = GFX_evt_large_army
	
	is_triggered_only = yes
	
	option = {
		name = "friendfavors.14.a"  # stop war
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				trait = just
			}
		}
		FROM = {
			letter_event = { id = friendfavors.15 tooltip = friend_given_peace_tooltip }
			opinion = {
				modifier = opinion_helpful_friend
				who = FROM
				years = 10
			}
			any_war = {
				limit = {
					any_defender = {character=ROOT}
				}
				any_attacker={letter_event = { id = friendfavors.16 tooltip = attacker_forced_peace_tooltip }}
			}
			
			any_war = {
				limit = {
					defender = {
						character = FROM
					}
					attacker = {
						same_realm = FROM
					}
				}
				end_war = invalid
			}
		}
		any_vassal = {
			limit = {
				NOT = { has_opinion_modifier = { who = ROOT modifier = opinion_friend } }
				 higher_tier_than = BARON
			}
			opinion = {
				modifier = opinion_abuse_power 
				who = ROOT
				years = 5
			}
		}
	}
	
	option = {
		name = "friendfavors.14.b"  # don't stop war
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				trait = arbitrary
			}
		}
		FROM = {
			letter_event = { id = friendfavors.17 tooltip = friend_given_war_tooltip }
			opinion = {
				modifier = opinion_unhelpful_friend
				who = ROOT
				years = 15
			}
		}
	}
}
