namespace = friendevents

# Events for making friends with the Make a Friend Ambition - by JGCrusader
# 0 - Councillor tries to befriend liege
# 1 - liege responds
# 2 - friends - liege and councillor 
# 3 - Sad - liege and councillor
# 4 - liege tries to befriend courtier
# 5 - courtier responds
# 6 - friends - courtier and liege
# 7 - Sad - courtier and liege
# 8 - Duelist Courtier offers liege friendship
# 9 - liege responds
# 10- Friends - liege and duelist
# 11- Sad - liege and duelist
# 12- Courtier offers another courtier friendship
# 13- other courtier responds
# 14- Friends - 2 courtiers
# 15- Sad - 2 courtiers

# Councillor tries to befriend liege
character_event = {
	id = friendevents.0
	desc = friendevents.0.desc
	picture = GFX_evt_courtiers_talking
	
	prisoner = no
	min_age = 25
	capable_only = yes
	only_men = yes
	
	trigger = {
		has_ambition = obj_make_a_friend
		in_command = no
		is_councillor = yes
		
		liege = {
			age = 25
			prisoner = no
			in_command = no
			is_female = no
			NOT = { trait = incapable }
			NOT = {
				num_of_friends = 2
			}
			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_friend }
			}
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_friend }
			}
			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_rival }
			}
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_rival }
			}
		}
		
		opinion = { who = LIEGE value = 10 } 

	}		

	mean_time_to_happen = {
		months = 140
		
		modifier = {
			factor = 0.85
			trait = gregarious
		}
		
		modifier = {
			factor = 0.65
			liege ={is_close_relative = ROOT}
		}
		
		modifier = {
			factor = 0.75
			liege = {
				tier = count
			}
		}
		
		modifier = {
			factor = 0.85
			liege = {
				tier = duke
			}
		}
		
		modifier = {
			factor = 0.85
			liege = {
				has_ambition = obj_make_a_friend
			}
		}
		
		modifier = {
			factor = 0.65
				opinion = { who = LIEGE value = 75 }
		}
	}
	
	option = {
		name = "friendevents.0.a"  # Let's talk to him
		ai_chance = { factor = 95 }
		liege = {
			character_event = {
				id = friendevents.1
				days = 1
				random = 3
				tooltip = "make_friend_attempt_tooltip"
			}
		}
	}
		
	option = {
		name = "friendevents.0.b"  # Maybe later
		ai_chance = { factor = 5 }
		random = {
				chance = 5
				add_trait = shy
			}
	}	
	
}

# ruler tries to befriend a courtier
character_event = {
	id = friendevents.4
	desc = friendevents.4.desc
	picture = GFX_evt_courtiers_talking
	
	prisoner = no
	min_age = 25
	capable_only = yes
	only_rulers = yes
	
	trigger = {
		has_ambition = obj_make_a_friend
		in_command = no
		
		any_courtier = {
			in_command = no
			is_female = no
			prisoner = no
			age = 16
			NOT = {
				num_of_friends = 2
			}
			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_friend }
			}
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_friend }
			}
			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_rival }
			}
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_rival }
			}
		}
		
		any_courtier = {
			is_female = yes
			age = 16
			prisoner = no
			NOT = {
				num_of_friends = 2
			}
			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_friend }
			}
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_friend }
			}
			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_rival }
			}
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_rival }
			}
		}
		
	}

	mean_time_to_happen = {
		months = 60
		
		modifier = {
			factor = 0.85
			trait = gregarious
		}
		
		modifier = {
			factor = 0.85
			age = 25
		}
		
		modifier = {
			factor = 0.85
			age = 35
		}
		
		modifier = {
			factor = 0.75
			lower_tier_than = duke
		}
	}
	
	option = {
		name = "friendevents.4.a"  # Let's talk to him
		ai_chance = { factor = 95 }
		# If male ruler, make friends with man
		if = {
			limit = {
				is_female = no
			}
			random_courtier = {
				limit = {
					in_command = no
					is_female = no
					prisoner = no
					age = 16
					NOT = {
						num_of_friends = 2
					}
					NOT = {
						has_opinion_modifier = { who = ROOT modifier = opinion_friend }
					}
					NOT = {
						reverse_has_opinion_modifier = { who = ROOT modifier = opinion_friend }
					}
					NOT = {
						has_opinion_modifier = { who = ROOT modifier = opinion_rival }
					}
					NOT = {
						reverse_has_opinion_modifier = { who = ROOT modifier = opinion_rival }
					}
				}
				
				character_event = {
					id = friendevents.5
					days = 1
					random = 3
					tooltip = "make_friend_attempt_tooltip"
				}
			}
		}
		
		# If female ruler, make friends with woman
		if = {
			limit = {
				is_female = yes
			}
			random_courtier = {
				limit = {
					is_female = yes
					age = 16
					prisoner = no
					NOT = {
						num_of_friends = 2
					}
					NOT = {
						has_opinion_modifier = { who = ROOT modifier = opinion_friend }
					}
					NOT = {
						reverse_has_opinion_modifier = { who = ROOT modifier = opinion_friend }
					}
					NOT = {
						has_opinion_modifier = { who = ROOT modifier = opinion_rival }
					}
					NOT = {
						reverse_has_opinion_modifier = { who = ROOT modifier = opinion_rival }
					}
				}
				
				character_event = {
					id = friendevents.5
					days = 1
					random = 3
					tooltip = "make_friend_attempt_tooltip"
				}
			}
		}
	}
		
	option = {
		name = "friendevents.4.b"  # Maybe later
		ai_chance = { factor = 5 }
		random = {
				chance = 5
				add_trait = shy
			}
	}	
	
}
