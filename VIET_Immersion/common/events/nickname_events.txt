character_event = {

	desc = nickevents.0.desc
	picture = "GFX_evt_comet"
	
	only_rulers = yes

	capable_only = yes
	prisoner = no

		OR={

		NOT = { trait = cruel }

			trait = kind

		OR = {
			diplomacy = 12 
			martial = 12 
			stewardship = 12

		}

		months = 130
