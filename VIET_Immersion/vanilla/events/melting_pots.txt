		year = 1080
		culture = norman

			OR = {

				culture = siculo_arabic
			}
			religion = ROOT
		}
	}
	
	mean_time_to_happen = {
		months = 200
		modifier = {
			factor = 0.75

				tier = count
			}
		}

			factor = 0.25
			primary_title = {
				tier = baron
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				tier = duke
			}
		}
		modifier = {
			factor = 5
			primary_title = {
				OR = {
					tier = king
					tier = emperor
				}
			}
		}
		modifier = {
			factor = 0.75
			NOT = { age = 10 }
		}
		modifier = {
			factor = 1.1
			age = 10
		}
		modifier = {
			factor = 1.1
			age = 16
		}
		modifier = {
			factor = 1.2
			age = 20
		}
		modifier = {
			factor = 1.35
			age = 30
		}
		modifier = {
			factor = 1.5
			age = 40
		}
		modifier = {
			factor = 2
			age = 50
		}
		modifier = {
			factor = 2
			age = 60
		}
		modifier = {
			factor = 0.75
			NOT = {
				capital_scope = {
					any_neighbor_province = {
						culture = ROOT
					}
				}
			}
		}
	}
	
	option = {
		name = "sicilian.meltingpot.a"
		ai_chance = { factor = 100 }
		culture = sicilian
		any_child = {
			limit = {
				dynasty = ROOT
				NOT = { age = 16 }
			}
			culture = sicilian
		}
	}
	option = {
		name = "sicilian.meltingpot.b"
	}
}
